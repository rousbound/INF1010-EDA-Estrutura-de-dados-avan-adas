 52 static Mapa* insere2(Mapa *m, int chave, int novodado, int *cresceu) {
 53   if (m == NULL) {
 54     m = cria_no(chave, novodado);
 55     *cresceu = 1;
 56   }
 57   else if (chave < m->chave) {
 58    m->esq = insere2(m->esq, chave, novodado, cresceu);
 59     if(*cresceu) {
 60       switch (m->bf) {
 61       case 1:
 62         m->bf = 0;
 63         *cresceu = 0;
 64         break;
 65       case 0:
 66         m->bf = -1;
 67         break;
 68       case -1:
 69         m = corrige_esq(m);
 70         *cresceu = 0;
 71         break;
 72       }
 73 
 74   }
 75     }
 76   else if (chave > m->chave) {
 77     m->dir = insere2(m->dir, chave, novodado, cresceu);
 78     if(*cresceu) {
 79       switch (m->bf) {
 80       case 1:
 81         m = corrige_dir(m);
 82         *cresceu = 0;
 83         break;
 84       case 0:
 85         m->bf = 1;
 86         break;
 87       case -1:
 88         m->bf = 0;
 89         *cresceu = 0;
 90         break;
 91       }
 92     }
 93   }
 94   return m;
 95 }

 98 static Mapa* corrige_esq(Mapa *m) {
 99   if (m->esq->bf == -1) {
100     m->bf = m->esq->bf = 0;
101     return rotaciona_dir(m);
102   }
103   else if (m->esq->bf == 1) {
104     Mapa *n = m->esq->dir;
105 
106     switch (n->bf) {
107     case -1:
108       m->bf = 1;
109       m->esq->bf = n->bf = 0;
110       break;
111     case 0:
112       m->bf = m->esq->bf = n->bf = 0;
113       break;
114     case 1:
115       m->bf = n->bf = 0;
116       m->esq->bf = -1;
117       break;
118     }
119     m->esq = rotaciona_esq(m->esq);
120     return rotaciona_dir(m);
121   }

172   if (m == NULL) {
173     return -1;
174   }
175   else if (m->chave == chave) {
176     return m->conteudo;
177   }
178   else if (chave < m->chave) {
179   return busca(m->esq, chave);
180   }
181   else if (chave > m->chave) {
182     return busca(m->dir, chave);
183   }
184   else {
185     return -1;
